using namespace std;
int n, m, a, b;
map<int, int> M;

int main(){
    ios::sync_with_stdio(0); cin.tie(0); cout.tie(0);
    cin >> n;
    for(int i=0; i<n; i++){
        cin >> a;
        auto it = M.find(a);
        if(it == M.end()) M[a] = 1;
        else              M[a] += 1;
    }
    
    cin >> m;
    for(int i=0; i<m; i++){
        cin >> b;
        auto it = M.find(b);
        if(it == M.end()) cout << 0 << ' ';
        else              cout << M[b] << ' ';
    }
    cout << '\n';
    return 0;
}
