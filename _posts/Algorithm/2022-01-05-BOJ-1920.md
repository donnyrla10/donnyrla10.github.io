#include <iostream>
#include <algorithm>
using namespace std;

int ps(int arr[], int x, int l, int r){
    while(l<=r){
        int mid = (l+r)/2;
        if(x == arr[mid]){
            return 1;
        }
        
        if(x > arr[mid]){
            l = mid+1;
        }else{
            r = mid-1;
        }
    }
    return 0;
}

int main(){
    ios::sync_with_stdio(0); cin.tie(0); cout.tie(0);
    int n, m, x, A[100002];
    cin >> n;
    for(int i=1; i <= n; i++) cin >> A[i];
    
    sort(A+1, A+1+n);
    cin >> m;
    for(int i=1; i <= m; i++){
        cin >> x;
        if (ps(A, x, 1, n)) cout << 1 << "\n";
        else cout << 0 << "\n";
    }
    return 0;
}
