//15656ë²ˆ: Nê³¼ M(7)
#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;
int n,m,a;
vector<int> v;
vector<int> ans;

void input(){
    cin >> n >> m;
    for(int i=0; i<n; i++){
        cin >> a;
        v.push_back(a);
    }
}

void rf(int k){
    if(m == k){
        for(auto it = ans.begin(); it != ans.end(); ++it)
            cout << *it << ' ';
        cout << '\n';
        return;
    }
    
    for(int i=0; i<n; i++){
        ans.push_back(v[i]);
        rf(k+1);
        ans.pop_back();
    }
}

int main(){
    ios::sync_with_stdio(0); cin.tie(0); cout.tie(0);
    input();
    sort(v.begin(), v.end());
    rf(0);
    return 0;
}
