#include <iostream>

int n, m, an[500001], am[500001];

void input(){
    cin >> n;
    for(int i=0; i<n; i++) cin >> an[i];

bool lower_bound(int x, int l, int r){

        }else if(an[mid] < x){
            l = mid + 1;

    }
    return false;
