//11399ë²ˆ: ATM
#include <iostream>
#include <algorithm>
using namespace std;
int n, arr[1001], sum = 0;

void input(){
    cin >> n;
    for(int i=0; i<n; i++)
        cin >> arr[i];
    sort(arr, arr+n); //ì •ë ¬
}

void rfunc(int k, int value){
    if(k == n){
        cout << sum << '\n';
        return;
    }
    sum += value;
    rfunc(k+1, value+arr[k+1]);
}

int main(){
    ios::sync_with_stdio(0); cin.tie(0); cout.tie(0);
    input();
    rfunc(0, arr[0]);
    return 0;
}
