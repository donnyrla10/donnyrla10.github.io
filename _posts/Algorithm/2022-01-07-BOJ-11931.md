#include <iostream>
#include <algorithm>
using namespace std;

bool compare(int a, int b){
    return a > b;
}

int main(){
    int n, arr[1000001];
    scanf("%d", &n);
    for(int i=0; i<n; i++) scanf("%d", &arr[i]);
    sort(arr, arr+n, compare);
    for(int i=0; i<n; i++) printf("%d\n", arr[i]);
    return 0;
}
