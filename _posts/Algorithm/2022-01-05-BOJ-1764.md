//1764ë²ˆ: ë“£ë³´ì¡
#include <iostream>
#include <algorithm>
using namespace std;
int n, m, cnt = 0;
string an[500001], am[500001], ans[500001];

void input(){
    cin >> n >> m;
    for(int i=0; i<n; i++) cin >> an[i];
    for(int i=0; i<m; i++) cin >> am[i];
}

void ps(string x, int l, int r){
    while(l <= r){
        int mid = (l+r)/2;
        if(x == am[mid]){
            ans[cnt++] = x;
            return;
        }else if(x > am[mid]){
            l = mid + 1;
        }else if(x < am[mid]){
            r = mid - 1;
        }
    }
}

int main(){
    ios::sync_with_stdio(0); cin.tie(0); cout.tie(0);
    input();
    
    sort(an, an+n);
    sort(am, am+m);
    
    for(int i=0; i<n; i++){
        ps(an[i], 0, m-1);
    }
    
    cout << cnt << '\n';
    for(int i=0; i<cnt; i++)
        cout << ans[i] << '\n';
    return 0;
}
